题目描述：

数组中有一个数字出现的次数超过数组长度的一半，请找出这个数字。例如输入一个长度为9的数组{1,2,3,2,2,2,5,4,2}。由于数字2在数组中出现了5次，超过数组长度的一半，因此输出2。如果不存在则输出0。

实现如下：

//方法一
//原理：数组中出现次数超过一半的数字，在其有序序列中就是中位数
//利用partition的作用：将某个数字，放到它应该在的位置，而将其他数分类摆放，比它小的放它前面，比它大的放它后面
//注意：
//1.测试用例中可能存在重复的数，所以必须在判断条件的上注意稳定性
//2.防御性动作
//3.测试用例中有可能不存在满足条件的数，即出现次数最多的数都没有超过一半
class Solution
{
public:
	int partition(vector<int>arr, int left, int right)
	{
		int tmp = arr[left];
		while (left < right)//直到left和right重合，此时合适位置找到
		{
			while (arr[right] >= tmp && left < right)//等于号考虑稳定性
			{
				right--;
			}
			arr[left] = arr[right];

			while (arr[left] <= tmp && left < right)//等于号考虑稳定性
			{
				left++;
			}
			arr[right] = arr[left];
		}
		arr[left] = tmp;//将tmp放在它应该在的位置
		return left;
	}
	bool checkNumber(vector<int> numbers, int number)//检查结果是否满足条件
	{
		int time = 0;//计数器
		for (int i = 0; i < numbers.size(); ++i)
		{
			if (numbers[i] == number)
				++time;
		}
		if (time*2 <= numbers.size())//判断出现的次数是否超过一半
			return false;
		else
			return true;
	}
	int MoreThanHalfNum_Solution(vector<int> numbers)
	{
		if (numbers.empty())//防御性动作
			return 0;
		int middle = numbers.size() / 2;
		int left = 0;
		int right = numbers.size() - 1;
		int pa = partition(numbers, left, right);//先进行一次partition
		while (middle != pa)//判断是否为中位数
		{
			if (pa < middle)//如果位置小于中位数，则对pa的右边partition
			{
				left = pa + 1;
				pa = partition(numbers, left, right);
			}
			else if (pa > middle)//如果位置大于中位数，则对pa的左边partition
			{
				right = pa - 1;
				pa = partition(numbers, left, right);
			}
		}
		if (checkNumber(numbers, numbers[middle]))//判断结果是否满足条件
			return numbers[middle];
		else
			return 0;
	}
};
//方法二
//方法二可以不用改变vector
//数组中出现次数超过一半的数字，其出现的次数一定比其他数出现次数总和还要多
//设置一个计数器time，等于res，++time；不等于res，若time==0，则替换res,计数器置1，否则--time
//注意：
//1.防御性动作
//2.判断结果是否满足条件
class Solution 
{
public:
	bool checkNumber(vector<int> numbers, int number)//判断结果是否满足条件
	{
		int time = 0;
		for (int i = 0; i < numbers.size(); ++i)
		{
			if (numbers[i] == number)
				++time;
		}
		if (time * 2 <= numbers.size())
			return false;
		else
			return true;
	}
	int MoreThanHalfNum_Solution(vector<int> numbers) 
	{
		if (numbers.empty())//防御性动作
			return 0;
		int time = 0;//计数器初始化为0
		int res = numbers[0];//记录结果
		for (int i = 0; i < numbers.size(); ++i)
		{
			if (res == numbers[i])//若与结果相同，计数器自加
				++time;
			else
			{
				if (time == 0)//此时res进行替换，将新res的time置为1
				{
					res = numbers[i];
					++time;
				}
				else//否则，计数自减
					--time;
			}
		}
		if (checkNumber(numbers, res))//判断结果是否满足条件
			return res;
		else
			return 0;
	}
};
